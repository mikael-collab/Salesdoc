function doGet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var result = ss.getSheets().map(function(sh) {
    return { name: sh.getName(), rows: sh.getDataRange().getValues() };
  });
  return ContentService
    .createTextOutput(JSON.stringify({ ok: true, sheets: result }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var p  = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sh = ss.getSheetByName(p.sheet);
    if (!sh) throw new Error('Sheet not found: ' + p.sheet);

    if (p.action === 'fillFirstEmpty') {
      // Find first row (after header) where the key column is empty
      var data = sh.getDataRange().getValues();
      var targetRow = -1;
      for (var r = 1; r < data.length; r++) {
        var cell = data[r][p.keyCol];
        if (cell === '' || cell === null || cell === undefined) {
          targetRow = r + 1; // 1-indexed
          break;
        }
      }
      if (targetRow === -1) {
        // No pre-existing empty row — append
        sh.appendRow(p.row);
      } else {
        sh.getRange(targetRow, 1, 1, p.row.length).setValues([p.row]);
      }
    } else if (p.action === 'append') {
      sh.appendRow(p.row);
    } else {
      sh.getRange(p.cell).setValue(p.value);
    }

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
