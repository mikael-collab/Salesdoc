// ⚠️ Replace PASTE_SHEET_ID_HERE with your actual Google Sheet ID
// Sheet ID is the long code in the URL:
// docs.google.com/spreadsheets/d/SHEET_ID/edit
var SHEET_ID = 'PASTE_SHEET_ID_HERE';

function doGet(e) {
  var ss = SpreadsheetApp.openById(SHEET_ID);

  // Write operation via ?data= parameter
  if (e && e.parameter && e.parameter.data) {
    return handleWrite(ss, JSON.parse(e.parameter.data));
  }

  // Read operation — return all sheet data
  var result = ss.getSheets().map(function(sh) {
    return { name: sh.getName(), rows: sh.getDataRange().getValues() };
  });
  return jsonResponse({ ok: true, sheets: result });
}

function doPost(e) {
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    return handleWrite(ss, JSON.parse(e.postData.contents));
  } catch(err) {
    return jsonResponse({ ok: false, error: err.message });
  }
}

function handleWrite(ss, p) {
  try {
    var sh = ss.getSheetByName(p.sheet);
    if (!sh) throw new Error('Sheet not found: ' + p.sheet);

    if (p.action === 'writeSparseRow') {
      var data = sh.getDataRange().getValues();
      var keyCol = typeof p.keyCol === 'number' ? p.keyCol : 1;
      var targetRow = -1;
      for (var r = 1; r < data.length; r++) {
        var cell = data[r][keyCol];
        if (cell === '' || cell === null || cell === undefined) {
          targetRow = r + 1;
          break;
        }
      }
      if (targetRow === -1) targetRow = data.length + 1;
      p.cells.forEach(function(c) {
        sh.getRange(targetRow, c.col + 1).setValue(c.val);
      });

    } else if (p.action === 'append') {
      sh.appendRow(p.row);

    } else if (p.cell) {
      sh.getRange(p.cell).setValue(p.value);
    }

    return jsonResponse({ ok: true });
  } catch(err) {
    return jsonResponse({ ok: false, error: err.message });
  }
}

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
