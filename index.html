⚙️ Manager / Admin setup guide

Do this once per salesperson. ~5 minutes. No Google Cloud needed.

  • Open the salesperson's Google Sheet → click Extensions → Apps Script
  • Delete all existing code. Paste the script below and click Save 💾
  • Click Deploy → New deployment
    • Type: Web app
    • Execute as: Me
    • Who has access: Anyone
    Click Deploy
  • Google asks you to authorise — click Review permissions → Allow. This is safe and normal.
  • Copy the Web app URL and send it to the salesperson. That's their personal form URL — they paste it in step 1 above.
function doGet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var result = ss.getSheets().map(function(sh) {
    return { name: sh.getName(), rows: sh.getDataRange().getValues() };
  });
  return ContentService
    .createTextOutput(JSON.stringify({ ok: true, sheets: result }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var p  = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sh = ss.getSheetByName(p.sheet);
    if (!sh) throw new Error("Sheet not found: " + p.sheet);
    sh.getRange(p.cell).setValue(p.value);
    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

💡 The sheet structure is read automatically. Column A = labels, Column B = editable values. Each sheet tab becomes a navigation tab in the app.
